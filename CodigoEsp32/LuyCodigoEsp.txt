void desligarServos() {

for (int ch = 0; ch < 16; ch++) {

pwm.setPWM(ch, 0, 4096);

}

Serial.println("Servos em repouso.");

}



void fecharMao() {

Serial.println("Comando: Fechar Mão (Limite Reduzido)");




for (int angle0 = 0; angle0 <= 140; angle0 += 10) {


int pulse0 = map(angle0, 0, 180, SERVOMAX, SERVOMIN);

int pulse6 = map(angle0, 0, 180, SERVOMIN, SERVOMAX);




pwm.setPWM(0, 0, pulse0); pwm.setPWM(3, 0, pulse0); pwm.setPWM(13, 0, pulse0);

pwm.setPWM(6, 0, pulse6); pwm.setPWM(9, 0, pulse6); pwm.setPWM(12, 0, pulse6);

pwm.setPWM(14, 0, pulse6); pwm.setPWM(15, 0, pulse6);




if (angle0 > 60) {

int angle12 = angle0 - 60;



int pulse1 = map(angle12, 0, 80, SERVOMIN, map(140, 0, 180, SERVOMIN, SERVOMAX));

int pulse2 = map(angle12, 0, 80, SERVOMAX, map(140, 0, 180, SERVOMAX, SERVOMIN));


pwm.setPWM(1, 0, pulse1); pwm.setPWM(4, 0, pulse1);

pwm.setPWM(7, 0, pulse1); pwm.setPWM(10, 0, pulse1);



pwm.setPWM(2, 0, pulse2); pwm.setPWM(5, 0, pulse2);

pwm.setPWM(8, 0, pulse2); pwm.setPWM(11, 0, pulse2);

}

delay(20);

}

desligarServos();

atualizarDisplay("Mao Fechada", "---");

}

Serial.println("Comando: Abrir Mão (Lógica Original)");



for (int angle0 = 180; angle0 >= 0; angle0 -= 10) {

// Servos da Base

pwm.setPWM(0, 0, pulse0);

}



void gestoJoinha() {

atualizarDisplay("Joinha!", "OK");

Serial.println("Gesto: Joinha");




for (int angle0 = 0; angle0 <= 140; angle0 += 10) {



int p_inv_val = map(angle0, 0, 180, SERVOMAX, SERVOMIN);

int p_nor_val = map(angle0, 0, 180, SERVOMIN, SERVOMAX);



// FECHANDO OS 4 DEDOS (Base)

pwm.setPWM(0, 0, p_inv_val); // Mindinho

pwm.setPWM(3, 0, p_inv_val); // Anelar

pwm.setPWM(6, 0, p_nor_val); // Médio

pwm.setPWM(9, 0, p_nor_val); // Indicador



// MANTENDO O DEDÃO ABERTO

pwm.setPWM(13, 0, map(0, 0, 180, SERVOMAX, SERVOMIN));

pwm.setPWM(14, 0, map(0, 0, 180, SERVOMIN, SERVOMAX));

pwm.setPWM(15, 0, map(0, 0, 180, SERVOMIN, SERVOMAX));





if (angle0 > 60) {

int angle12 = angle0 - 60; // Máximo será 80 (140 - 60)



// Mapeamento reduzido para as pontas não forçarem o plástico

int p1 = map(angle12, 0, 80, SERVOMIN, map(140, 0, 180, SERVOMIN, SERVOMAX));

int p2 = map(angle12, 0, 80, SERVOMAX, map(140, 0, 180, SERVOMAX, SERVOMIN));


// Articulações do Mindinho, Anelar, Médio e Indicador

pwm.setPWM(1, 0, p1); pwm.setPWM(2, 0, p2);

pwm.setPWM(4, 0, p1); pwm.setPWM(5, 0, p2);

pwm.setPWM(7, 0, p1); pwm.setPWM(8, 0, p2);

pwm.setPWM(10, 0, p1); pwm.setPWM(11, 0, p2);

}

delay(20);

}

desligarServos();

}

class MyCallbacks : public BLECharacteristicCallbacks {

void onWrite(BLECharacteristic *pCharacteristic) {

// CORREÇÃO: Pegamos o valor diretamente como String do Arduino

String value = pCharacteristic->getValue();



if (value.length() > 0) {

int ssidIndex = value.indexOf("SSID:");

int passIndex = value.indexOf(";PASS:");


if (ssidIndex >= 0 && passIndex >= 0) {

String ssid = value.substring(ssidIndex + 5, passIndex);

String pass = value.substring(passIndex + 6);



ssid.trim();

pass.trim();


Serial.println("--- Nova configuração recebida ---");

Serial.print("SSID Extraído: ["); Serial.print(ssid); Serial.println("]");



WiFi.disconnect(true, true);

delay(500);


WiFi.begin(ssid.c_str(), pass.c_str());

Serial.println("Iniciando conexão WiFi...");

} else {

Serial.println("Formato BLE inválido. Use SSID:nome;PASS:senha");

}

}

}

};

void handleCommand() {

server.sendHeader("Access-Control-Allow-Origin", "*");

String tipo = server.arg("tipo");


if (tipo == "abrir") abrirMao();

else if (tipo == "fechar") fecharMao();

else if (tipo == "joinha") gestoJoinha();

else if (tipo == "vezinho") gestoVezinho();



server.send(200, "text/plain", "OK");

}



void setup() {

Serial.begin(115200);

pinMode(LED_PIN, OUTPUT);

Wire.begin(21, 22);



// --- INICIALIZAÇÃO DO DISPLAY ---

if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {

Serial.println("Erro no OLED");

} else {

atualizarDisplay("Sistema", "ON");

}

////////////-----------------------------------//////////////





pwm.begin();

pwm.setPWMFreq(50);

desligarServos();



BLEDevice::init("MaoRobotica_Config");

BLEServer *pServer = BLEDevice::createServer();

BLEService *pService = pServer->createService(SERVICE_UUID);

pRxCharacteristic = pService->createCharacteristic(CHARACTERISTIC_UUID_RX, BLECharacteristic::PROPERTY_WRITE);

pRxCharacteristic->setCallbacks(new MyCallbacks());

pTxCharacteristic = pService->createCharacteristic(CHARACTERISTIC_UUID_TX, BLECharacteristic::PROPERTY_NOTIFY);

pService->start();

pServer->getAdvertising()->start();



// =====================================================

// BLOCO DE CONEXÃO CORRIGIDO

// =====================================================



WiFi.mode(WIFI_STA);

WiFi.disconnect(true);

delay(500);

WiFi.begin("Yurinha2g", "9295631045");

// =====================================================



server.on("/executar", handleCommand);

server.begin();

}

void loop() {

server.handleClient();

static bool connected = false;

if (WiFi.status() == WL_CONNECTED) {

digitalWrite(LED_PIN, HIGH);

if (!connected) {

Serial.print("Conectado! IP: "); Serial.println(WiFi.localIP());

MDNS.begin("maorobotica");

connected = true;

}

} else {

digitalWrite(LED_PIN, (millis() / 500) % 2);

connected = false;

}

}
